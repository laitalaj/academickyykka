1   package org.kyykka.logic;

3   import java.util.HashSet;
4   import java.util.Set;
5   import org.kyykka.logic.object.Kyykka;
6   
7   /**
8    * Contains a set of kyykkas for a team. Handles some interaction with kyykkas.
9    *
10   * @author Julius Laitala
11   */
12  public class KyykkaContainer {
13  
14      private Set<Kyykka> kyykkas;
15  
16      public KyykkaContainer(boolean homeTeam) {
17          this.kyykkas = new HashSet<>();
18          int y;
19          if (homeTeam) {
20              y = 15000;
21          } else {
22              y = 5000;
23          }
24          for (int z = 0; z <= 10; z += 10) {
25              for (int x = 125; x < 5000; x += 250) {
26                  this.kyykkas.add(new Kyykka(x, y, z));
27              }
28          }
29      }
30  
31      public Set<Kyykka> getKyykkas() {
32          return kyykkas;
33      }
34  
35      public void tick() {
36          //TODO: Kyykkien poistuminen pelialueelta, etc.
37          for (Kyykka k : kyykkas) {
38              k.tick();
39          }
40      }
41  
42  }
