3   import org.kyykka.logic.shape.Point;
4   
5   /**
6    * Thrower is the guy throwing the karttu, the character playing kyykka.
7    *
8    * @author Julius Laitala
9    */
10  public class Thrower extends PhysicsEntity {
11  
12      private Point target;
13  
14      /**
15       * Creates a thrower with specified position
16       *
17       * @param x x-position of the thrower
18       * @param y y-position of the thrower
19       */
20      public Thrower(int x, int y) {
21          super(x, y, 0, 300, 300, 1700, 80000);
22          this.target = new Point(x, y, 0);
23          this.setFrozen(false);
24      }
25  
26      /**
27       * Sets the coordinates to which the thrower tries to move
28       *
29       * @param x x-coordinate to be moved to
30       * @param y y-coordinate to be moved to
31       */
32      public void setTarget(int x, int y) {
33          this.target.setX(x);
34          this.target.setY(y);
35      }
36  
37      /**
38       * Sets the coordinates to which the thrower tries to move
39       *
40       * @param p point to be moved to, z-coordinate should be 0
41       */
42      public void setTarget(Point p) {
43          this.target = p;
44      }
45  
46      /**
47       * Calculates the speed with which the thrower should move towards its
48       * target (the thrower slows down when close to the point)
49       *
50       * @return the speed with which the thrower should move
51       */
52      public int calculateNextSpeed() {
53          Point center = this.getBoundingBox().getBottomCenter();
54          int distance = center.getDistance(this.target);
55          if (distance >= 3000) {
56              return 40;
57          } else {
58              double speed = 40 * ((double) distance / 3000);
59              if (speed < 5) {
60                  speed = 5;
61              }
62              if (distance < speed) {
63                  speed = distance;
64              }
65              return (int) speed;
66          }
67      }
68  
69      /**
70       * Updates the thrower's speed according to it's distance to its target
71       *
72       * @see org.kyykka.logic.object.Thrower#calculateNextSpeed()
73       */
74      public void updateSpeed() {
75          int vx = this.target.getX() - this.getX();
76          int vy = this.target.getY() - this.getY();
77          if (vx == 0 && vy == 0) {
78              this.setXmom(0);
79              this.setYmom(0);
80              return;
81          }
82          int nextspeed = calculateNextSpeed();
83          if (vx == 0) {
84              this.setYmom(nextspeed * (int) Math.signum(vy));
85              this.setXmom(0);
86          } else {
87              double ratio = (double) Math.abs(vy) / Math.abs(vx);
88              double xmom = (nextspeed / Math.sqrt(ratio * ratio + 1)) * Math.signum(vx);
89              //Calculations are basically Pythagorean theorem
90              double ymom = Math.abs(xmom) * ratio * Math.signum(vy);
91              this.setXmom((int) xmom);
92              this.setYmom((int) ymom);
93          }
94      }
95  
96      /**
97       * Creates a karttu at the throwers position with given initial angle and
98       * force
99       *
100      * @param angle angle of the throw in degrees (0 = straight ahead)
101      * @param force velocity of the throw in mm / cs
102      *
103      * @return the karttu that was thrown
104      */
105     public Karttu throwKarttu(int angle, int force) {
106         double angleradians = Math.toRadians(angle);
107         double xmom = force * Math.sin(angleradians);
108         double ymom = force * Math.cos(angleradians);
109         Point throwpos = this.getBoundingBox().getCenter();
110         return new Karttu(throwpos.getX(), throwpos.getY(), throwpos.getZ(), (int) xmom, (int) ymom, 0);
111     }
112 
113     /**
114      * Creates a karttu at the throwers position with given initial parameters
115      *
116      * @param p the parameters of the throw
117      *
118      * @see org.kyykka.logic.object.ThrowParams
119      *
120      * @return the karttu that was thrown
121      */
122     public Karttu throwKarttu(ThrowParams p) {
123         return this.throwKarttu(p.getAngle(), p.getForce());
124     }
125 
126     /**
127      * Override of PhysicsEntity's setFrozen; always sets frozen to false
128      *
129      * @param frozen does nothing, necessary for the override
130      */
131     @Override
132     public void setFrozen(boolean frozen) {
133         super.setFrozen(false);
134     }
135 
136     /**
137      * Updates the thrower by one tick. Updates its speed and moves it.
138      *
139      * @see org.kyykka.logic.object.Thrower#updateSpeed()
140      * @see org.kyykka.logic.object.PhysicsEntity#move()
141      */
142     @Override
143     public void tick() {
144         updateSpeed();
145         //TODO: Override PhysicsEntity move with one that doesn't slide
146         move();
147     }
148 
149     public Point getPos() {
150         return this.getBoundingBox().getBottomCenter();
151     }
152 
153 }
