3   /**
4    * Kyykka is a small, wooden piece that one is supposed to knock away with a
5    * karttu. The Kyykka class is a very barebones copy of PhysicsEntity
6    *
7    * @author Julius Laitala
8    */
9   public class Kyykka extends PhysicsEntity {
10  
11      /**
12       * Creates a kyykka with a specific position.
13       *
14       * @param x x-position of the kyykka
15       * @param y y-position of the kyykka
16       * @param z z-position of the kyykka
17       */
18      public Kyykka(int x, int y, int z) {
19          super(x, y, z, 80, 80, 100, 100);
20      }
21  
22      /**
23       * Updates the kyykka by one physics tick. Applies gravity and moves the
24       * kyykka.
25       *
26       * @see org.kyykka.logic.object.PhysicsEntity#applyGravity()
27       * @see org.kyykka.logic.object.PhysicsEntity#move()
28       */
29      @Override
30      public void tick() {
31          this.applyGravity();
32          this.move();
33      }
34  
35  }
