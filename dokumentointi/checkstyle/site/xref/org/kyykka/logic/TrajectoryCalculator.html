6   /**
7    * Trajectory calculator contains functions with which one can calculate where a
8    * PhysicsEntity with certain parameters will land.
9    *
10   * @author Admin
11   */
12  public class TrajectoryCalculator {
13  
14      /**
15       * Calculates the point where given PhysicsEntity will land.
16       *
17       * @param e the physics entity which trajectory should be calculated
18       *
19       * @return landing poisition
20       */
21      public static Point3D calculateLanding(PhysicsEntity e) {
22          double d = e.getZmom() * e.getZmom() + 2 * e.getZ();
23          if (d < 0) {
24              return null;
25          }
26          d = Math.sqrt(d);
27          double time = -e.getZmom() - d;
28          time *= -1;
29  //        double t2 = -e.getZmom() - D;
30          int x = e.getX() + e.getXmom() * (int) time;
31          int y = e.getY() + e.getYmom() * (int) time;
32          return new Point3D(x, y, 0);
33      }
34  }
