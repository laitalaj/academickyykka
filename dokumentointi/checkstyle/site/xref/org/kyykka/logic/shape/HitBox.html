7   /**
8    * A 3D box that handles positions and collision detection.
9    *
10   * @author Julius Laitala
11   */
12  public class HitBox {
13  
14      //The location represents lower bottom left corner
15      //(Smallest x, y and z of the box)
16      private Point3D location;
17  
18      private int width;
19      private int height;
20      private int depth;
21  
22      /**
23       * Creates a hitbox with the specified parameters.
24       *
25       * @param x x-position of the boxes lower bottom left corner
26       * @param y y-position of the boxes lower bottom left corner
27       * @param z z-position of the boxes lower bottom left corner
28       * @param width x-axis dimension of the box
29       * @param height y-axis dimension of the box
30       * @param depth z-axis dimension of the box
31       */
32      public HitBox(int x, int y, int z, int width, int height, int depth) {
33          this.location = new Point3D(x, y, z);
34          this.width = width;
35          this.height = height;
36          this.depth = depth;
37      }
38  
39      /**
40       * Creates a hitbox with the specified parameters.
41       *
42       * @param location lower bottom left corner of the box
43       * @param width x-axis dimension of the box
44       * @param height y-axis dimension of the box
45       * @param depth z-axis dimension of the box
46       */
47      public HitBox(Point3D location, int width, int height, int depth) {
48          this(location.getX(), location.getY(), location.getZ(),
49                  width, height, depth);
50      }
51  
52      /**
53       * Returns all corners of the box.
54       *
55       * @return list of corners
56       */
57      public Collection<Point3D> getCorners() {
58          Collection<Point3D> corners = new HashSet<>();
59          int[] xcoords = new int[]{this.getX(), this.getX() + this.width};
60          int[] ycoords = new int[]{this.getY(), this.getY() + this.height};
61          int[] zcoords = new int[]{this.getZ(), this.getZ() + this.depth};
62          for (int x : xcoords) {
63              for (int y : ycoords) {
64                  for (int z : zcoords) {
65                      corners.add(new Point3D(x, y, z));
66                  }
67              }
68          }
69          return corners;
70      }

72      /**
73       * Checks if the box collides with a point (if the point is inside the box).
74       *
75       * @param x x-coordinate of the point
76       * @param y x-coordinate of the point
77       * @param z x-coordinate of the point
78       *
79       * @return true if collision happens, false otherwise
80       */
81      public boolean collidesWith(int x, int y, int z) {
82          boolean collides = x >= this.location.getX() && x <= this.location.getX() + this.width;
83          collides = collides && y >= this.location.getY() && y <= this.location.getY() + this.height;
84          collides = collides && z >= this.location.getZ() && z <= this.location.getZ() + this.depth;
85          return collides;
86      }
87  
88      /**
89       * Checks if the box collides with a point (if the point is inside the box).
90       *
91       * @param p point with which to check collision
92       *
93       * @return true if collision happens, false otherwise
94       */
95      public boolean collidesWith(Point3D p) {
96          return collidesWith(p.getX(), p.getY(), p.getZ());
97      }
98  
99      /**
100      * Checks if the box collides with any point in a collection (if any of the
101      * points is inside the box).
102      *
103      * @param points points to check for collisions
104      *
105      * @return true if any point collides with the box, false otherwise
106      */
107     public boolean collidesWith(Collection<Point3D> points) {
108         for (Point3D p : points) {
109             if (this.collidesWith(p)) {
110                 return true;
111             }
112         }
113         return false;
114     }
115 
116     /**
117      * Checks if this box collides with another box (if either of the boxes
118      * contain parts of each other).
119      *
120      * @param box the box with which to check for collisions
121      *
122      * @return true if collision happens, false otherwise
123      */
124     public boolean collidesWith(HitBox box) {
125         boolean collides = this.getX() <= box.getX() + box.getWidth();
126         collides = collides && this.getX() + this.width >= box.getX();
127         collides = collides && this.getY() <= box.getY() + box.getHeight();
128         collides = collides && this.getY() + this.height >= box.getY();
129         collides = collides && this.getZ() <= box.getZ() + box.getDepth();
130         collides = collides && this.getZ() + this.depth >= box.getZ();
131         return collides;
132     }
133 
134     /**
135      * Checks if this box collides with any of the boxes in given collection.
136      *
137      * @param boxes boxes to be checked collision with
138      *
139      * @return true if collision happens, false otherwise
140      */
141     public boolean collidesWithAny(Collection<HitBox> boxes) {
142         for (HitBox b : boxes) {
143             if (this.collidesWith(b)) {
144                 return true;
145             }
146         }
147         return false;
148     }

150     /**
151      * Calculates the center position of the box.
152      *
153      * @return the center position
154      */
155     public Point3D getCenter() {
156         int x = this.getX() + this.width / 2;
157         int y = this.getY() + this.height / 2;
158         int z = this.getZ() + this.depth / 2;
159         return new Point3D(x, y, z);
160     }
161 
162     /**
163      * Sets the boxes center position to given point.
164      *
165      * @param point point to set center to
166      */
167     public void setCenter(Point3D point) {
168         this.setX(point.getX() - this.width / 2);
169         this.setY(point.getY() - this.height / 2);
170         this.setZ(point.getZ() - this.depth / 2);
171     }
172 
173     /**
174      * Calculates the upper bottom left position of the box (x and z are equal
175      * to position, y is position + height).
176      *
177      * @return the upper bottom left position
178      */
179     public Point3D getUpperBottomLeft() {
180         return new Point3D(this.getX(), this.getY() + this.height, this.getZ());
181     }
182 
183     /**
184      * Calculates the bottom center position of the box (center on x and y axis,
185      * low z-axis).
186      *
187      * @return the bottom center position
188      */
189     public Point3D getBottomCenter() {
190         Point3D point = this.getCenter();
191         point.setZ(this.getZ());
192         return point;
193     }
194 
195     /**
196      * Calculates the top center position of the box (center on x and y axis,
197      * high z-axis).
198      *
199      * @return the top center position
200      */
201     public Point3D getTopCenter() {
202         Point3D point = this.getCenter();
203         point.setZ(this.getZ() + this.depth);
204         return point;
205     }

207     /**
208      * Calculates the lower top left position of the box (x and y are equal to
209      * position, z is position + depth).
210      *
211      * @return the lower top left position
212      */
213     public Point3D getLowerTopLeft() {
214         return new Point3D(this.getX(), this.getY(), this.getZ() + this.depth);
215     }
216 
217     /**
218      * Calculates the lower top left position of the box (x is equal to
219      * position, z is position + depth, y is (position + height) / 2).
220      *
221      * @return the center top left position
222      */
223     public Point3D getCenterTopLeft() {
224         Point3D topCenter = getTopCenter();
225         topCenter.setX(this.getX());
226         return topCenter;
227     }

333     @Override

339     @Override
340     public int hashCode() {
341         int hash = 3;
342         hash = 67 * hash + Objects.hashCode(this.location);
343         hash = 67 * hash + this.width;
344         hash = 67 * hash + this.height;
345         hash = 67 * hash + this.depth;
346         return hash;
347     }

349     @Override
350     public boolean equals(Object obj) {
351         if (obj == null) {
352             return false;
353         }
354         if (getClass() != obj.getClass()) {
355             return false;
356         }
357         final HitBox other = (HitBox) obj;
358         if (!Objects.equals(this.location, other.location)) {
359             return false;
360         }
361         if (this.width != other.width) {
362             return false;
363         }
364         if (this.height != other.height) {
365             return false;
366         }
367         if (this.depth != other.depth) {
368             return false;
369         }
370         return true;
371     }
