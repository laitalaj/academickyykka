17  /**
18   * The Main class of Academic Kyykka World Cup. Handles the interaction between
19   * all other classes.
20   *
21   * @author Julius Laitala
22   */
23  public class Main {
24  
25      /**
26       * The main method. This is where the magic starts. Creates a new game,
27       * loads images, initializes display and runs everything.
28       *
29       * @param args command line arguments (do nothing at the moment)
30       */
31      public static void main(String[] args) {
32          //Works!
33          ImageContainer container = new ImageContainer();
34          Game game = new Game();
35          GamePanel gamePainter = new GamePanel(1200, 700, game, container);
36          MenuPanel menu = new MenuPanel();
37          EndPanel end = new EndPanel();
38          Map<String, JPanel> panels = new HashMap<>();
39          panels.put("menu", menu);
40          panels.put("game", gamePainter);
41          panels.put("end", end);
42          Display display = new Display(panels, "menu");
43          SwingUtilities.invokeLater(display);
44          Semaphore lock = new Semaphore(1);
45          while (true) { //Main loop
46              GameInitializer init = new GameInitializer(display, menu, lock);
47              lock.acquireUninterruptibly();
48              lock.release();
49              game = init.getGame();
50              display.switchPanel("game");
51              gamePainter.setGame(game);
52              gamePainter.setActive(true);
53              game.setWinningTeam(game.getActiveTeam());
54              game.run();
55              gamePainter.setActive(false);
56              display.switchPanel("end");
57              EndHandler endHandler = new EndHandler(game.getWinningTeam(), end,
58                      lock);
59              lock.acquireUninterruptibly();
60              lock.release();
61              display.switchPanel("menu");
62          }
63  //        System.exit(0);
64      }
65  }
