1   package org.kyykka.graphics;
2   
3   /**
4    * Interface for classes that can be drawn. Used for interaction between
5    * PhysicsEntities, GamePainter and ImageContainer
6    *
7    * @author Julius Laitala
8    */
9   public interface Drawable {
10  
11      /**
12       * Should return the name of the image file that is wanted to represent the
13       * drawable object. The image file must be present in the img folder.
14       *
15       * @return image name with file extension (not path)
16       */
17      public String getImgName();
18  
19      /**
20       * Should return the amount of transparency the image is to be drawn with. 1
21       * means fully opaque, 0 is completely transparent.
22       *
23       * @return amount of transparency
24       */
25      public float getAlpha();
26  }
